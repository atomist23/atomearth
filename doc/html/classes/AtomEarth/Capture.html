     # File lib/atomearth/capture.rb, line 47
 47:     def start
 48:       # Start capturing
 49:       rd, wr = IO.pipe
 50: 
 51:       # Lets fork
 52:       @capfork = fork do
 53:         Signal.trap("TERM") { exit }
 54:         
 55:         rd.close
 56: 
 57:         # Get capture device from config
 58:         httpdump = Pcaplet.new CONFIG.device
 59: 
 60:         # Filter for capture
 61:         filter = Pcap::Filter.new 'tcp and dst port 80', httpdump.capture
 62:         httpdump.add_filter filter
 63: 
 64:         # This is the main capture loop analyzing the packets
 65:         httpdump.each_packet do |pkt|
 66:           data = pkt.tcp_data
 67:           next unless data
 68:           request_line, data = data.split("\r\n", 2)
 69:           next unless data
 70:           next unless request_line =~ /^GET\s+(\S+)/
 71:           path = $1
 72:           next unless path =~ /.htm|.jsp|.asp|.cfm|.php|\/$/
 73:           headers = data.split("\r\n\r\n", 2)
 74:           next unless headers.first
 75:           headers = headers.first.split("\r\n")
 76:           headers = headers.map { |line| line.split(': ', 2) }.flatten
 77:           next unless headers.length % 2 == 0
 78:           headers = Hash[*headers]
 79: 
 80:           host = headers['Host'] || pkt.dst.to_s
 81:           host << ":#{pkt.dst_port}" if pkt.dport != 80
 82: 
 83:           # And here we are sending the data to the capture thread for insertion into the buffer
 84:           wr.puts [is_intnet(pkt.src.to_num_s), pkt.src.to_s, pkt.src.to_num_s, host, pkt.dst.to_num_s, path].inspect
 85:         end      
 86:         wr.close     
 87:       end
 88: 
 89:       wr.close
 90: 
 91:       @capthread = Thread.start do
 92:         # Capture thread. This will receive data from the fork and insert it into the urls buffer
 93:         @alive = true
 94:         until rd.eof? do
 95:           line = rd.gets
 96:           break if line.nil?
 97:           @urls << eval(line)
 98:           @count.add
 99:         end
100:       end
101: 
102:       @updatethread = Thread.start do
103:         while @capthread.alive? do
104:           @count.update
105:           sleep 1
106:         end
107:         Process.kill("TERM", @capfork.to_i)
108:         @count.update
109:         @alive = false
110:       end
111:     end
